    <div class="container">

        <h1>Simple Calculator</h1>

        <p>Enter a number (0-999):</p>

        <input type="number" id="inputField" min="0" max="999">

        <button onclick="calculateResult()">Calculate</button>

        <p id="result"></p>

    </div>


    <script>

        function calculateResult() {

            const inputField = document.getElementById('inputField');

            const resultField = document.getElementById('result');

            const X = parseInt(inputField.value);


            // Check if the input is valid

            if (isNaN(X) || X < 0 || X > 999) {

                resultField.textContent = 'Please enter a valid number between 0 and 999.';

            } else {

                const Y = X + 100;

                resultField.textContent = `The result is: ${Y}`;

            }

        }

    </script>
